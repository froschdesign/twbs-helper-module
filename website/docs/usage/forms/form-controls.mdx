# Form controls

[Twitter bootstrap Documentation](https://getbootstrap.com/docs/5.1/forms/form-control)

## Example

[Twitter bootstrap Documentation](https://getbootstrap.com/docs/5.1/forms/form-control/#example)

<Tabs>
<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.1">
      <div dangerouslySetInnerHTML={{ __html: `<form action="" method="POST" name="form" role="form" id="form">
    <div class="mb-3">
        <label class="form-label" for="exampleFormControlInput1">Email address</label>
        <input name="email" type="email" id="exampleFormControlInput1" placeholder="name@example.com" class="form-control" value=""></input>
    </div>
    <div class="mb-3">
        <label class="form-label" for="exampleFormControlTextarea1">Example textarea</label>
        <textarea name="textarea" id="exampleFormControlTextarea1" rows="3" class="form-control"></textarea>
    </div>
</form>` }}></div>
  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

$oFactory = new \Laminas\Form\Factory();

echo $this->form(
    $oFactory->create(
        [
            'type'     => 'form',
            'elements' => [
                [
                    'spec' => [
                        'name'       => 'email',
                        'options'    => ['label' => 'Email address'],
                        'attributes' => [
                            'type'        => 'email',
                            'id'          => 'exampleFormControlInput1',
                            'placeholder' => 'name@example.com',
                        ],
                    ],
                ],
                [
                    'spec' => [
                        'name'       => 'textarea',
                        'options'    => ['label' => 'Example textarea'],
                        'attributes' => [
                            'type' => 'textarea',
                            'id'   => 'exampleFormControlTextarea1',
                            'rows' => 3,
                        ],
                    ],
                ],
            ],
        ]
    )
);

```

</TabItem>

<TabItem value="result" label="Result" default>
  <HtmlCode bootstrapVersion="5.1">
      <div dangerouslySetInnerHTML={{ __html: `<input type="text" name="lg" placeholder=".form-control-lg" aria-label=".form-control-lg example" class="form-control form-control-lg" value=""></input>
<br></br>
<input type="text" name="default" placeholder="Default input" aria-label="default input example" class="form-control" value=""></input>
<br></br>
<input type="text" name="sm" placeholder=".form-control-sm" aria-label=".form-control-sm example" class="form-control form-control-sm" value=""></input>` }}></div>
  </HtmlCode>
</TabItem>
<TabItem value="source" label="Source">

```php
<?php

$oFactory = new \Laminas\Form\Factory();

// Render large input
$oElement = $oFactory->create(
    [
        'name'       => 'lg',
        'type'       => 'text',
        'options'    => ['size' => 'lg'],
        'attributes' => [
            'placeholder' => '.form-control-lg',
            'aria-label'  => '.form-control-lg example',
        ],
    ]
);
echo $this->formElement($oElement);

echo PHP_EOL.'<br/>';

// Render default input
$oElement = $oFactory->create(
    [
        'name'       => 'default',
        'type'       => 'text',
        'attributes' => [
            'placeholder' => 'Default input',
            'aria-label'  => 'default input example',
        ],
    ]
);
echo $this->formElement($oElement);

echo PHP_EOL.'<br/>';

// Render small input
$oElement = $oFactory->create(
    [
        'name'       => 'sm',
        'type'       => 'text',
        'options'    => ['size' => 'sm'],
        'attributes' => [
            'placeholder' => '.form-control-sm',
            'aria-label'  => '.form-control-sm example',
        ],
    ]
);
echo $this->formElement($oElement);

```

```php
<?php

$oFactory = new \Laminas\Form\Factory();

    [
        'name'       => 'disabled-input',
        'type'       => 'text',
        'attributes' => [
            'disabled'    => true,
            'placeholder' => 'Disabled input',
            'aria-label'  => 'Disabled input example',
        ],
    ]

echo PHP_EOL.'<br/>';

    [
        'name'       => 'disabled-readonly-input',

        'attributes' => [
            'disabled'    => true,
            'readonly'    => true,
            'placeholder' => 'Disabled readonly input',
            'aria-label'  => 'Disabled input example',
        ],
    ]
);
echo $this->formElement($oElement);
```

```php
        'elements' => [
            [
                'spec' => [
                    'name'       => 'email',
                    'options'    => [
                        'plaintext' => true,
                        'column'    => 'sm-10',
                        'label'     => 'Email',

                        'label'  => 'Password',
                    ],
                    'attributes' => [
                        'type' => 'password',
                        'id'   => 'inputPassword',
                    ],
                ],
            ],
        ],
    ]

echo $this->form($oForm);

echo PHP_EOL.'<br/>';

// Render inline form

    $oFactory->create(
        [
            'type'     => 'form',

                [
                    'spec' => [
                        'name'       => 'email',

                        'attributes' => [

                    ],
```
